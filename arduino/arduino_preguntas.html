JUEGO DE PREGUNTAS

Arduino realiza preguntas, que contestamos a traves de pulsadores,
arduino nos dice si hemos acertado y al final del programa el numero de preguntas, acertadas.


   

Interes:
  • Funciones
  • Comparación de variable tipo caracter "char"
  • Monitor serial como salida.
  • Instrucción While

 Preguntas.

Realizamos una función para resolver.



int ledverde=12;

int ledrojo=10;

int pulsador1=2;
int pulsador2=3;
int pulsador3=4;

int aciertos=0;

char respuesta;
char solucion;

  
void setup(){
 pinMode(ledverde,OUTPUT);
  pinMode(ledrojo,OUTPUT);

  pinMode(pulsador1,INPUT);
 pinMode(pulsador2,INPUT);
 pinMode(pulsador3,INPUT);

  Serial.begin(9600);  
}
void resuelve(){
   while ( digitalRead(pulsador1)==LOW && digitalRead(pulsador2)==LOW && digitalRead(pulsador3)==LOW)
 {}
 if (digitalRead(pulsador1)== HIGH) respuesta='A';
if (digitalRead(pulsador2)==HIGH) respuesta='B';
if (digitalRead(pulsador3)==HIGH) respuesta='C';

if (respuesta==solucion){
Serial.println ("correcta");
digitalWrite (ledverde,HIGH);
delay (1000);
digitalWrite (ledverde,LOW);
delay (1000);
digitalWrite (ledverde,HIGH);
delay (1000);
digitalWrite (ledverde,LOW);
aciertos++;
}


if (respuesta!=solucion){
Serial.println ("fallo");
digitalWrite (ledrojo,HIGH);
delay (1000);
digitalWrite (ledrojo,LOW);
delay (1000);
digitalWrite (ledrojo,HIGH);
delay (1000);
digitalWrite (ledrojo,LOW);
}
}



  void loop(){
Serial.println("Cual es la capital de Espana");
Serial.println("A: Madrid");
Serial.println("B: Lisboa");
Serial.println("C:Dublin");

solucion='A';
resuelve();
 

Serial.println("Como se llaman la palabras que se pronuncia igual pero se escriben de forma distinta");
Serial.println("A: Sinonimas");
Serial.println("B: Polisemicas");
Serial.println("C: las dos respuestas son mentita");

solucion='C';
resuelve(); 
 
 
 




Serial.print("Has acertado ");
Serial.print (aciertos);
Serial.println (" respuestas ");
Serial.println ("Si quieres jugar otra vez pulsa una tecla");
while ( digitalRead(pulsador1)==LOW && digitalRead(pulsador2)==LOW && digitalRead(pulsador3)==LOW)
 {}
 aciertos=0;
 for (int n=0;n<10;n++){
  Serial.println(); // escribo 10 lineas en blanco simulando un borrado de pantalla
 }
 delay(500);
}
